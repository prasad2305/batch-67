Azure App Service Plans
Resources
| where type == "microsoft.web/serverfarms"
| extend zoneRedundant = tostring(properties.zoneRedundant)
| project name, location, zoneRedundant, sku = sku.name, tier = sku.tier, kind, resourceGroup, subscriptionId


Azure SQL MI

resources
| where type == "microsoft.sql/managedinstances/databases"
| extend dbName = name,
         miId = tostring(split(id, "/databases/")[0])
| join kind=inner (
    resources
    | where type == "microsoft.sql/managedinstances"
    | extend failoverPartnerId = tostring(properties.partnerManagedInstanceId)
    | project miName = name,
              miLocation = location,
              miSku = sku.name,
              miTier = sku.tier,
              miId = id,
              miResourceGroup = resourceGroup,
              miSubscriptionId = subscriptionId,
              failoverPartnerId
) on miId
| join kind=leftouter (
    resourcecontainers
    | where type == "microsoft.resources/subscriptions"
    | project miSubscriptionId = id,
              subscriptionDisplayName = name
) on miSubscriptionId
| project 
    subscriptionDisplayName,
    miSubscriptionId,
    miResourceGroup,
    miName,
    dbName,
    miLocation,
    miSku,
    miTier,
    failoverPartnerId
| order by subscriptionDisplayName, miName, dbName
